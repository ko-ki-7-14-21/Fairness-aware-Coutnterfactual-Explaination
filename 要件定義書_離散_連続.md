seed_data=[0,1,2,3,4]
alpha=0.25
beta=0.25
limit_1=240
limit_2=60
n_list=[5,25,125]
rho_list=[0.5,1.0,2.0]
K_list=[2,10,25]
epsilon=0.005
L=10
r=(1/2*epsilon)**(1/L)
gamma_list=[0]+[1+epsilon*(r ** l),1-epsilon*(r ** l) for l in range(L)]
J=len(gamma_list)

for seed in seed_data:
    for K in K_list:
        for n in n_list:
            for rho in rho_list:
                m=n*rho
                if n =5:
                    size = "small"
                elif n=25:
                    size="middle"
                else:
                    size="big"
                A:m*n の行列
                b[k](k=1,...,K):m次元ベクトル
                c_hat:n次元ベクトル
                A,b[k],c_hatの要素:0.1以上1.0以下の閉区間の一様分布からランダムに選択
                次の問題(LP)を解く
                    \min_{x_hat[1],...,x_hat[K]} \sum^K_{k=1} c_hat@x_hat[k]
                    s.t. A@x_hat[k]>=b[k], x[k]>=0 (k=1,...,K)
                問題LPの解をX_hat=[x_hat[1],...,x_hat[K]]とする
                h:n次元ベクトル
                h[i]=\sum^K_{k=1} x_hat[k][i](i=1,...,n)
                h[i]=!0であるようなiをn/3個ランダムに選択し、その要素iについてh[i]=0.8*h[i]
                その他の要素i'はh[i']=\max{10,10*h[i']}とする
                c_barを次のように作成
                c_bar=[gamma * c_hat for gamma in gamma_list]
                次の問題(D-D-FACE)を解く(計算時間上限=limit_1)
                \min_{c,v,v_ave,v_max,X,Y,Z} (1-alpha-beta) * (1/n) \sum^n_{i=1}(100*(c[i]-c_hat[i])/c_hat[i])**2 + alpha * (1/K) \sum^K_{k=1}(v[k]-v_ave) ** 2 + beta * v_max ** 2
                s.t. c[i]=\sum^J_{j=1}c_bar[i][j]*z[i][j], z[i][j]\in{0,1}, \sum^J_{j=1} z[i][j]=1,
                    A@x[k] >= b[k], x[k]>= 0
                    A.T@y[k] =< c, y[k]>= 0
                    c@x[k] = b@y[k]
                    v[k] = 100 * c@x[k] / c_hat@x_hat[k]
                    v_max >= v[k]
                    v_max >= 0
                    \sum^K_{k=1} x[k] =< h
                print(f"D-D-FACEP Done seed={seed}, K={K}, n={n}, m={m},status={(D-D-FACE)のstatus},t={(D-D-FACE)の計算時間})
                問題(D-D-FACE)の解X=[x[1],...,x[K]]を使って次の問題(D-FACE)を解く(計算時間上限=limit_2)
                \min_{c,v,v_ave,v_max,Y} (1-alpha-beta) * (1/n) \sum^n_{i=1}(100*(c[i]-c_hat[i])/c_hat[i])**2 + alpha * (1/K) \sum^K_{k=1}(v[k]-v_ave) ** 2 + beta * v_max ** 2
                s.t. A.T@y[k] =< c, y[k]>= 0
                    c@x[k] = b@y[k]
                    v[k] = 100 * c@x[k] / c_hat@x_hat[k]
                    v_max >= v[k]
                    v_max >= 0
                計算された解c,X=[x[1],...,x[K]]から次を計算する
                v[k] = 100 * c@x[k] / c_hat@x_hat[k]
                v_max = max{0,v[1],...,v[K]}
                price_term=\sum^n_{i=1}(100*(c[i]-c_hat[i])/c_hat[i])**2
                fairness_term1=\sum^K_{k=1}(v[k]-v_ave) ** 2
                fairness_term2=v_max**2
                objective = (1-alpha-beta) * (1/n) * price_term + alpha * (1/K) fairness_term1 + beta * fairness_term2
                G = (1/n)*price_term + (1/K) * fairness_term1 + fairness_term2
                csvファイル("random_D_C.csv")に次の内容を出力
                [K, m, n, size, objective, 問題(D-D-FACE)の計算時間とgapとstatus, 問題(D-FACE)の計算時間とgapとstatus, 評価指標]
                print(f"D-FACE Done seed={seed}, K={K}, n={n}, m={m},status={(D-FACE)のstatus},t={(D-FACE)の計算時間})
