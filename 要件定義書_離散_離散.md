seed_data=[0,1,2,3,4]
alpha=0.25
beta=0.25
limit_1=150
limit_2=150
n_list=[5,25,125]
rho_list=[0.5,1.0,2.0]
K_list=[2,10,25]
L_1=5
r_1=0.9/L_1
L_2=7
r__2=r_1/L_2
gamma_list_1=[0]+[r_1 * l, -r_1 * l for l in range(1,L_1+1)]
gamma_list_2=[0]+[r_2 * l, -r_2 * l for l in range(1,L_2+1)]
J_1=len(gamma_list_1)
J_2=len(gamma_list_2)

for seed in seed_data:
    for K in K_list:
        for n in n_list:
            for rho in rho_list:
                m=n*rho
                if n =5:
                    size = "small"
                elif n=25:
                    size="middle"
                else:
                    size="big"
                A:m*n の行列
                b[k](k=1,...,K):m次元ベクトル
                c_hat:n次元ベクトル
                A,b[k],c_hatの要素:0.1以上1.0以下の閉区間の一様分布からランダムに選択
                次の問題(LP)を解く
                    \min_{x_hat[1],...,x_hat[K]} \sum^K_{k=1} c_hat@x_hat[k]
                    s.t. A@x_hat[k]>=b[k], x[k]>=0 (k=1,...,K)
                問題LPの解をX_hat=[x_hat[1],...,x_hat[K]]とする
                h:n次元ベクトル
                h[i]=\sum^K_{k=1} x_hat[k][i](i=1,...,n)
                h[i]=!0であるようなiをn/3個ランダムに選択し、その要素iについてh[i]=0.8*h[i]
                その他の要素i'はh[i']=\max{10,10*h[i']}とする
                c_bar_1を次のように作成
                c_bar_1=[(1 + gamma) * c_hat for gamma in gamma_list_1]
                次の問題(D-D-FACE_1)を解く(計算時間上限=limit_1,Numeric_Focus=1)
                \min_{c,v,v_ave,v_max,X,Y,Z} (1-alpha-beta) * (1/n) \sum^n_{i=1}(100*(c[i]-c_hat[i])/c_hat[i])**2 + alpha * (1/K) \sum^K_{k=1}(v[k]-v_ave) ** 2 + beta * v_max ** 2
                s.t. c[i]=\sum^J_{j=1}c_bar_1[i][j]*z[i][j], z[i][j]\in{0,1}, \sum^J_{j=1} z[i][j]=1,
                    A@x[k] >= b[k], x[k]>= 0
                    A.T@y[k] =< c, y[k]>= 0
                    c@x[k] = b@y[k]
                    v[k] = 100 * c@x[k] / c_hat@x_hat[k]
                    v_max >= v[k]
                    v_max >= 0
                    \sum^K_{k=1} x[k] =< h
                print(f"D-D-FACEP_1 Done seed={seed}, K={K}, n={n}, m={m},status={(D-D-FACE_1)のstatus},t={(D-D-FACE_1)の計算時間})
                問題(D-D-FACE_1)の解c^*を使って、c_bar_2を次のように作成
                c_bar_2=[c^* + (gamma) * c_hat for gamma in gamma_list_2]
                次の問題(D-D-FACE_2)を解く(計算時間上限=limit_2,Numeric_Focus=1)
                \min_{c,v,v_ave,v_max,X,Y,Z} (1-alpha-beta) * (1/n) \sum^n_{i=1}(100*(c[i]-c_hat[i])/c_hat[i])**2 + alpha * (1/K) \sum^K_{k=1}(v[k]-v_ave) ** 2 + beta * v_max ** 2
                s.t. c[i]=\sum^J_{j=1}c_bar_2[i][j]*z[i][j], z[i][j]\in{0,1}, \sum^J_{j=1} z[i][j]=1,
                    A@x[k] >= b[k], x[k]>= 0
                    A.T@y[k] =< c, y[k]>= 0
                    c@x[k] = b@y[k]
                    v[k] = 100 * c@x[k] / c_hat@x_hat[k]
                    v_max >= v[k]
                    v_max >= 0
                    \sum^K_{k=1} x[k] =< h
                print(f"D-D-FACEP_2 Done seed={seed}, K={K}, n={n}, m={m},status={(D-D-FACE_2)のstatus},t={(D-D-FACE_2)の計算時間})
                計算された解c,X=[x[1],...,x[K]]から次を計算する
                v[k] = 100 * c@x[k] / c_hat@x_hat[k]
                v_max = max{0,v[1],...,v[K]}
                price_term=\sum^n_{i=1}(100*(c[i]-c_hat[i])/c_hat[i])**2
                fairness_term1=\sum^K_{k=1}(v[k]-v_ave) ** 2
                fairness_term2=v_max**2
                objective = (1-alpha-beta) * (1/n) * price_term + alpha * (1/K) fairness_term1 + beta * fairness_term2
                G = (1/n)*price_term + (1/K) * fairness_term1 + fairness_term2
                csvファイル("random_D_D.csv")に次の内容を出力
                [K, m, n, size, objective, 問題(D-D-FACE_1)の計算時間とgapとstatus, 問題(D-D-FACE_2)の計算時間とgapとstatus, 評価指標]
